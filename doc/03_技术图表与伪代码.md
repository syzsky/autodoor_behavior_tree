**防止假卡死的关键设计：**

| 组件 | 作用 | 注意事项 |
|------|------|---------|
| `_task_queue` | 线程安全队列，存储待处理的 UI 更新任务 | 使用 `Queue`，不使用 `list` |
| `_poll()` | 主线程轮询，每 30ms 检查队列 | **必须在编辑器初始化时启动** |
| `_poll_daemon()` | 后台线程守护，当 `_widget` 为 None 时继续尝试 | 防止轮询永久停止 |
| `_node_status_cache` | 合并相同节点的重复状态更新 | 减少不必要的重绘 |

### 2.5 WindowManager 伪代码

```
Algorithm: EnumAllWindows
Input: None
Output: List[Tuple[int, str]] - 窗口句柄和标题列表

function enum_all_windows():
    user32 = ctypes.windll.user32
    windows = []
    
    // 定义回调函数
    function enum_windows_proc(hwnd, lparam):
        if user32.IsWindowVisible(hwnd):
            length = user32.GetWindowTextLengthW(hwnd)
            if length > 0:
                buffer = create_unicode_buffer(length + 1)
                user32.GetWindowTextW(hwnd, buffer, length + 1)
                title = buffer.value
                if title:
                    windows.append((hwnd, title))
        return True
    
    // 枚举窗口
    WNDENUMPROC = WINFUNCTYPE(BOOL, HWND, LPARAM)
    user32.EnumWindows(WNDENUMPROC(enum_windows_proc), 0)
    
    return windows


Algorithm: FindWindowByTitle
Input: keyword (str) - 窗口标题关键字
Output: Optional[int] - 窗口句柄

function find_window_by_title(keyword):
    windows = enum_all_windows()
    keyword_lower = keyword.lower()
    
    for hwnd, title in windows:
        if keyword_lower in title.lower():
            return hwnd
    
    return None


Algorithm: SwitchToWindow
Input: hwnd (int) - 窗口句柄
Output: bool - 是否成功

function switch_to_window(hwnd):
    user32 = ctypes.windll.user32
    
    // 恢复窗口（如果最小化）
    SW_RESTORE = 9
    user32.ShowWindow(hwnd, SW_RESTORE)
    
    // 设置窗口位置（置顶）
    SWP_NOMOVE = 0x0002
    SWP_NOSIZE = 0x0001
    SWP_SHOWWINDOW = 0x0040
    user32.SetWindowPos(hwnd, -1, 0, 0, 0, 0, SWP_NOMOVE | SWP_NOSIZE | SWP_SHOWWINDOW)
    
    // 设置前台窗口
    return user32.SetForegroundWindow(hwnd) != 0


Algorithm: FindWindowByPid
Input: pid (int) - 进程ID
Output: Optional[int] - 窗口句柄

function find_window_by_pid(pid):
    results = []
    
    WNDENUMPROC = WINFUNCTYPE(BOOL, HWND, LPARAM)
    
    function enum_windows_callback(hwnd, _):
        if user32.IsWindowVisible(hwnd):
            window_pid = DWORD()
            user32.GetWindowThreadProcessId(hwnd, byref(window_pid))
            if window_pid.value == pid:
                length = user32.GetWindowTextLengthW(hwnd)
                if length > 0:
                    buffer = create_unicode_buffer(length + 1)
                    user32.GetWindowTextW(hwnd, buffer, length + 1)
                    if buffer.value:
                        results.append(hwnd)
        return True
    
    user32.EnumWindows(WNDENUMPROC(enum_windows_callback), 0)
    
    if results:
        return results[0]
    return None


Algorithm: GetWindowPid
Input: hwnd (int) - 窗口句柄
Output: Optional[int] - 进程ID

function get_window_pid(hwnd):
    try:
        pid = DWORD()
        user32.GetWindowThreadProcessId(hwnd, byref(pid))
        return pid.value if pid.value else None
    except Exception:
        return None


Algorithm: FindWindowSmart
Input: pid (Optional[int]) - 进程ID
       title_keyword (str) - 窗口标题关键字
Output: Tuple[Optional[int], str] - (窗口句柄, 查找方式)

function find_window_smart(pid, title_keyword):
    // 优先通过PID查找
    if pid:
        hwnd = find_window_by_pid(pid)
        if hwnd:
            return (hwnd, "pid")
    
    // PID查找失败，通过标题关键字查找
    if title_keyword:
        hwnd = find_window_by_title(title_keyword)
        if hwnd:
            return (hwnd, "title")
    
    return (None, "not_found")
```

```
function get_screenshot(region=None):
    if self._bound_window:
        // 后台截图模式
        from window_capture import WindowCapture
        from coordinate import CoordinateConverter
        
        if region:
            // 转换为窗口相对坐标

            )
            return WindowCapture.capture_window_region(self._bound_window, window_region)

        // 原有截图逻辑

Algorithm: SmartSwitchToBoundWindow
Input: None
Output: bool - 是否成功

function smart_switch_to_bound_window():
    if not self._bound_window:
        return False
    
    from window_manager import WindowManager
    
    // 已经是前台窗口，无需切换
    if WindowManager.is_foreground_window(self._bound_window):
        return True
    
    // 保存当前前台窗口
    self._saved_foreground_window = WindowManager.get_foreground_window()
    
    // 切换到绑定窗口
    return WindowManager.switch_to_window(self._bound_window)


Algorithm: SmartRestoreForegroundWindow
Input: None
Output: bool - 是否成功

function smart_restore_foreground_window():
    if not self._bound_window or not self._saved_foreground_window:
        return False
    
    from window_manager import WindowManager
    
    // 如果保存的窗口不是绑定窗口，则恢复
    if self._saved_foreground_window != self._bound_window:
        return WindowManager.switch_to_window(self._saved_foreground_window)
    
    return True
```

```
function _bind_window_to_context(context):
    from window_manager import WindowManager
    from log_manager import LogManager
    
    // 智能查找：优先PID，其次标题
    hwnd, find_method = WindowManager.find_window_smart(
        pid=self.window_pid if self.window_pid > 0 else None,
        title_keyword=self.window_title
    )
    
    if hwnd:
        // 绑定窗口
        context.bind_window(hwnd)
        
        // 获取窗口信息
        rect = WindowManager.get_window_rect(hwnd)
        title = WindowManager.get_window_title(hwnd)
        actual_pid = WindowManager.get_window_pid(hwnd)
        
        if find_method == "pid":
            print(f"[DEBUG] StartNode 通过PID绑定窗口: pid={self.window_pid}, hwnd={hwnd}, title='{title}', rect={rect}")
        else:
            print(f"[DEBUG] StartNode 通过标题绑定窗口: title='{self.window_title}', hwnd={hwnd}, actual_title='{title}', pid={actual_pid}, rect={rect}")
            if actual_pid and actual_pid != self.window_pid:
                print(f"[DEBUG] StartNode 提示: 窗口PID已变更 ({self.window_pid} -> {actual_pid})，建议重新选择窗口")
    else:
        LogManager.instance().log_failure(
            node_type="开始节点",
            node_name=self.name,
            reason=f"未找到窗口: pid={self.window_pid}, title='{self.window_title}'"
        )
```

```
function __init__(node_id, config):
    super().__init__(node_id, config)
    self._window_switched = False


Algorithm: ActionNodeTick
Input: context (ExecutionContext)
Output: NodeStatus

function _tick_internal(context):
    bound_window = context.get_bound_window()
    
    // 检查是否需要窗口切换（绑定了窗口且节点不跳过窗口切换）
    if bound_window and not self.SKIP_WINDOW_SWITCH:
        // 首次执行时切换窗口
        if not self._window_switched:
            print(f"[DEBUG] ActionNode '{self.name}' 检测到绑定窗口: hwnd={bound_window}")
            context.smart_switch_to_bound_window()
            self._window_switched = True
        
        print(f"[DEBUG] ActionNode '{self.name}' 执行动作...")
        status = self._execute_action(context)
        
        // 动作完成时恢复原窗口（RUNNING状态不恢复）
        if status != NodeStatus.RUNNING:
            context.smart_restore_foreground_window()
            self._window_switched = False
        
        self.status = status
    else:
        // 无窗口绑定或跳过窗口切换，直接执行
        status = self._execute_action(context)
    
    // 子节点执行逻辑
    // ...
    
    return status
```

```
        // 保存原始坐标用于日志
        original_region = region
        
        // 转换为窗口相对坐标
        region = CoordinateConverter.screen_region_to_window(region, bound_window)
        
        print(f"[DEBUG] RegionField: 坐标转换 屏幕绝对{original_region} -> 窗口相对{region}, hwnd={bound_window}")
    
    // 更新输入框
    self.entry.delete(0, "end")
    self.entry.insert(0, f"{region[0]}, {region[1]}, {region[2]}, {region[3]}")
    self._on_change(self._key, self.entry.get())
```

```
Algorithm: MagnifierWindow
State: zoom_factor, size, window, canvas, info_label

function show(x, y):
    if window is not None:
        return
    
    // 创建无边框顶层窗口
    window = create_toplevel()
    window.overrideredirect(true)
    window.attributes('-topmost', true)
    
    // 创建画布
    canvas = create_canvas(window, width=size, height=size)
    canvas.pack()
    
    // 创建信息标签
    info_label = create_label(
        window,
        font="Consolas 9",
        bg="#2b2b2b",
        fg="#ffffff",
        anchor="w",
        padx=5,
        pady=2,
        justify="left",
        highlightthickness=0
    )
    info_label.pack(fill="x")
    
    update(x, y)

function update(x, y):
    if window is None:
        return  // 必须先调用 show()
    
    // 计算截图区域
    half_size = size / (2 * zoom_factor)
    capture_region = (
        x - half_size, y - half_size,
        x + half_size, y + half_size
    )
    
    // 截取屏幕并放大
    screenshot = grab_screen(capture_region)
    zoomed_image = screenshot.resize(
        (size * zoom_factor, size * zoom_factor)
    )
    
    // 显示放大图像
    canvas.display_image(zoomed_image)
    
    // 绘制十字准线（红色虚线）
    center = size * zoom_factor / 2
    canvas.create_line(center, 0, center, size * zoom_factor, 
                       fill="#ff0000", dash=(4, 4), width=1)  // 垂直线
    canvas.create_line(0, center, size * zoom_factor, center, 
                       fill="#ff0000", dash=(4, 4), width=1)  // 水平线
    
    // 获取颜色信息
    color = get_pixel_color(x, y)
    r, g, b = color
    hex_color = format("#{:02X}{:02X}{:02X}", r, g, b)
    
    // 更新信息显示（换行显示）
    info_text = format("坐标: ({}, {})\n颜色: {}\nRGB({}, {}, {})", 
                       x, y, hex_color, r, g, b)
    info_label.config(text=info_text)
    
    // 更新窗口位置（鼠标右下方）
    window.geometry(format("+{}+{}", x + 20, y + 20))

function hide():
    if window is not None:
        window.destroy()
        window = None
        canvas = None
        info_label = None

function set_selection_rect(x1, y1, x2, y2):
    if _selection_rect is not None:
        canvas.delete(_selection_rect)
    
    _selection_rect = canvas.create_rectangle(
        x1, y1, x2, y2,
        outline="#000000",  // 黑色边框
        width=2
    )
```

```
function _select_region():
    magnifier = MagnifierWindow(zoom_factor=4, size=150)
    magnifier_shown = [false]
    start_pos = [null]
    
    // 绑定全局鼠标事件
    function on_mouse_move(event):
        if not magnifier_shown[0]:
            magnifier.show(event.x_root, event.y_root)
            magnifier_shown[0] = true

            magnifier.update(event.x_root, event.y_root)

        unbind_events()
        
        if start_pos[0] is not None:
            x1, y1 = start_pos[0]
            x2, y2 = event.x_root, event.y_root
            region = (min(x1,x2), min(y1,y2), max(x1,x2), max(y1,y2))
            set_value(region)
            on_change()
    
    bind_event('<Motion>', on_mouse_move)
    bind_event('<ButtonPress-1>', on_mouse_press)
    bind_event('<ButtonRelease-1>', on_mouse_release)
```

```
function get_random_value(base_value, random_range=0, min_value=null):
    // 1. 检查是否需要随机
    if random_range <= 0:
        return base_value
    
    // 2. 计算范围边界
    min_val = base_value - random_range
    max_val = base_value + random_range
    
    // 3. 应用最小值限制
    if min_value is not None:
        min_val = max(min_val, min_value)
    
    // 4. 返回随机值
    return random.randint(int(min_val), int(max_val))
```

### 3.2 Sequence 节点 Tick 算法

```
Algorithm: SequenceTick
Input: children[], current_index, context
Output: NodeStatus

function sequence_tick(children, current_index, context):
    if children is empty:
        return SUCCESS
    
    while current_index < len(children):
        child = children[current_index]
        
        // 跳过禁用节点
        if not child.config.enabled:
            current_index += 1
            continue
        
        status = child.tick(context)
        
        switch status:
            case RUNNING:
                return RUNNING    // 暂停，等待下次tick
            case FAILURE:
                current_index = 0 // 重置索引
                return FAILURE    // 短路返回
            case SUCCESS:
                current_index += 1  // 继续下一个
        
    // 所有子节点成功
    current_index = 0
    return SUCCESS
```

### 3.3 Parallel 节点 Tick 算法

```
Algorithm: ParallelTick
Input: children[], cached_statuses, success_policy, context
Output: NodeStatus

function parallel_tick(children, cached_statuses, success_policy, context):
    if children is empty:
        log_success(node_type="并行节点", node_name=self.name)
        return SUCCESS
    
    success_count = 0
    failure_count = 0
    running_children = []
    enabled_count = 0
    
    for i, child in enumerate(children):
        if not child.config.enabled:
            continue
        enabled_count += 1
        
        // 跳过已完成的子节点
        if i in cached_statuses:
            status = cached_statuses[i]
            if status == SUCCESS: success_count += 1; continue
            if status == FAILURE: failure_count += 1; continue
        
        // Tick 未完成的子节点
        status = child.tick(context)
        
        switch status:
            case SUCCESS:
                cached_statuses[i] = SUCCESS
                success_count += 1
            case FAILURE:
                cached_statuses[i] = FAILURE
                failure_count += 1
            case RUNNING:
                running_children.append(child)
    
    // require_one 策略：任一成功即返回
    if success_policy == REQUIRE_ONE and success_count > 0:
        abort_running(running_children)
        log_success(node_type="并行节点", node_name=self.name)
        return SUCCESS
    
    // 仍有运行中的子节点
    if running_children is not empty:
        return RUNNING
    
    // 所有子节点执行完毕
    if success_policy == REQUIRE_ALL:
        if success_count == enabled_count:
            log_success(node_type="并行节点", node_name=self.name)
            return SUCCESS
        else:
            log_failure(
                node_type="并行节点",
                node_name=self.name,
                reason=f"成功 {success_count}/{enabled_count} 个子节点"
            )
            return FAILURE
    else:
        if success_count > 0:
            log_success(node_type="并行节点", node_name=self.name)
            return SUCCESS
        else:
            log_failure(
                node_type="并行节点",
                node_name=self.name,
                reason=f"成功 {success_count}/{enabled_count} 个子节点"
            )
            return FAILURE
```

### 3.3.1 StartNode 节点 Tick 算法

```
Algorithm: StartNodeTick
Input: children[], current_index, context
Output: NodeStatus

function start_node_tick(children, current_index, context):
    if children is empty:
        log_success(node_type="开始节点", node_name=self.name)
        return SUCCESS
    
    has_failure = false
    
    while current_index < len(children):
        child = children[current_index]
        
        // 跳过禁用节点
        if not child.config.enabled:
            current_index += 1
            continue
        
        status = child.tick(context)
        
        switch status:
            case RUNNING:
                return RUNNING    // 等待当前子节点完成
            case FAILURE:
                has_failure = true
                current_index += 1  // 继续执行下一个子节点
            case SUCCESS:
                current_index += 1  // 继续执行下一个子节点
    
    // 所有子节点执行完毕，重置索引
    current_index = 0
    
    if has_failure:
        log_failure(
            node_type="开始节点",
            node_name=self.name,
            reason="部分子节点执行失败"
        )
        return FAILURE
    else:
        log_success(node_type="开始节点", node_name=self.name)
        return SUCCESS
```

**关键实现细节：**
- `current_index` 记录当前执行位置，支持 RUNNING 状态的记忆
- 子节点失败时设置 `has_failure = true`，但继续执行后续子节点
- 所有子节点执行完毕后重置 `current_index = 0`
- 跳过 `enabled = False` 的子节点

### 3.3.2 ActionNode 子节点状态追踪算法

```
Algorithm: ActionNodeTick
Input: _children_running, children[], context
Output: NodeStatus

function action_node_tick(_children_running, children, context):
    // 检查子节点是否正在执行
    if _children_running and children is not empty:
        return execute_children(context)
    
    // 执行动作
    status = execute_action(context)
    self.status = status
    
    // 动作成功后执行子节点
    if status == SUCCESS and children is not empty:
        // 通知 GUI 节点状态为 SUCCESS
        context.notify_node_status(self.node_id, "success")
        _children_running = true
        return execute_children(context)
    
    return status

function execute_children(context):
    status = SUCCESS
    
    for child in children:
        if not child.config.enabled:
            continue
        
        child_status = child.tick(context)
        
        if child_status == RUNNING:
            return RUNNING
        elif child_status == FAILURE:
            status = FAILURE
    
    // 子节点执行完成，重置状态
    _children_running = false
    return status
```

### 3.3.3 ConditionNode 子节点状态追踪算法

```
Algorithm: ConditionNodeTick
Input: _children_running, children[], context, check_interval_ms
Output: NodeStatus

function condition_node_tick(_children_running, children, context):
    // 检查子节点是否正在执行
    if _children_running and children is not empty:
        return execute_children(context)
    
    // 检测间隔控制
    current_time = context.elapsed_time * 1000
    if current_time - last_check_time < check_interval_ms:
        return self.status
    
    last_check_time = current_time
    
    // 执行条件检测
    result = check_condition(context)
    
    if invert:
        result = not result
    
    status = SUCCESS if result else FAILURE
    self.status = status
    
    // 条件成功后执行子节点
    if status == SUCCESS and children is not empty:
        // 通知 GUI 节点状态为 SUCCESS
        context.notify_node_status(self.node_id, "success")
        _children_running = true
        return execute_children(context)
    
    return status
```

**关键实现细节：**
- `_children_running` 标志避免在子节点运行期间重复执行动作/条件检测
- 在执行子节点前必须通知 GUI 节点状态为 SUCCESS
- 子节点执行完成后重置 `_children_running = False`

### 3.3.4 图像匹配算法（含置信度返回）

```
Algorithm: ImageTemplateMatch
Input: source_image, template_image, threshold
Output: Tuple[found: bool, position: Optional[Tuple], confidence: float]

function find_template(source, template, threshold):
    // 转换为 OpenCV 格式
    source_cv = cv2.cvtColor(np.array(source), cv2.COLOR_RGB2BGR)
    template_cv = cv2.cvtColor(np.array(template), cv2.COLOR_RGB2BGR)
    
    // 模板匹配
    result = cv2.matchTemplate(source_cv, template_cv, cv2.TM_CCOEFF_NORMED)
    
    // 获取最大匹配值（置信度）
    min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(result)
    
    // 判断是否匹配成功
    if max_val >= threshold:
        // 计算中心点坐标
        h, w = template_cv.shape[:2]
        center_x = max_loc[0] + w // 2
        center_y = max_loc[1] + h // 2
        return (True, (center_x, center_y), max_val)
    
    // 匹配失败，返回最高置信度用于日志输出
    return (False, None, max_val)

// 使用示例
function image_condition_check(context):
    found, position, confidence = find_template(screenshot, template, threshold)
    
    if found:
        context.blackboard.set(position_key, position)

        return True
    else:
        log_failure(
            node_type="图像检测节点",
            node_name=self.name,
            reason=f"未找到匹配的图像，最高置信度: {confidence:.2%}，阈值: {threshold:.2%}"
        )
        return False
```

### 3.4 撤销/重做算法

```
Algorithm: CommandManager
State: undo_stack, redo_stack, max_history=100, is_executing=false

function execute(command):
    if is_executing: return false
    is_executing = true
    try:
        if command.execute():
            undo_stack.push(command)
            redo_stack.clear()
            if len(undo_stack) > max_history:
                undo_stack.pop_oldest()
            return true
        return false
    finally:
        is_executing = false

function undo():
    if undo_stack is empty: return false
    is_executing = true
    try:
        command = undo_stack.pop()
        if command.undo():
            redo_stack.push(command)
            return true
        undo_stack.push(command)  // 恢复失败，放回
        return false
    finally:
        is_executing = false

function redo():
    if redo_stack is empty: return false
    is_executing = true
    try:
        command = redo_stack.pop()
        if command.redo():
            undo_stack.push(command)
            return true
        redo_stack.push(command)  // 重做失败，放回
        return false
    finally:
        is_executing = false
```

```
        _retry_count = 0
        _repeat_count = 0

    _tick_count = 0
    _start_time = None

    _tick_count = 0
    _start_time = None

    reset(reset_counters=false)  // 不重置计数器

    status = RUNNING  // 关键：必须是 RUNNING
```

```
    with _pool_lock:
        if node_id in _executor_pool:
            executor = _executor_pool[node_id]
            if executor.is_running:
                return executor  // 返回正在运行的执行器

        _executor_pool[node_id] = executor
        return executor

    with _lock:
        // 停止当前执行器
        if _executor and _executor.is_running:
            _executor.stop_script()
        
        // 从执行器池中移除
        with _pool_lock:
            if node_id in _executor_pool:
                delete _executor_pool[node_id]
        
        _executor = None
        _script_started = False
        _aborted = False
    
    // 调用父类重置
    super.reset(reset_counters=reset_counters)

// 执行脚本
function _execute_action(context):
    executor = _get_or_create_executor()
    
    // 循环执行逻辑：loop=true 且 repeat_count=0
    use_loop = loop and config.repeat_count == 0
    
    if not _script_started:
        _script_started = True
        executor.run_script(_script_content, loop=use_loop)
    
    if executor.is_running:
        return RUNNING
    elif executor.last_error:
        return FAILURE
    else:
        return SUCCESS
```

### 3.7 ScriptExecutor delay 命令非阻塞算法

```
Algorithm: NonBlockingDelay
State: is_running, current_command

function execute_delay_command(command):
    delay_time = command["time"] / 1000  // ms → s
    elapsed = 0
    
    // 分段检查 is_running，支持中断
    while elapsed < delay_time and is_running:
        sleep_time = min(0.1, delay_time - elapsed)  // 最多睡 100ms
        sleep(sleep_time)
        elapsed += sleep_time
    
    // 如果被中断，返回 false
    if not is_running:
        return false
    
    return true
```

**设计要点：**
- 将长时间 delay 分解为多个 100ms 的小段
- 每段结束后检查 `is_running` 标志
- 用户点击停止时，最多等待 100ms 即可响应

### 3.11 自动布局算法

```
Algorithm: AutoLayout
Input: nodes_data, root_id
Output: positions Dict[node_id → (x, y)]

const NODE_WIDTH = 180
const NODE_HEIGHT = 80
const H_GAP = 50
const V_GAP = 100

function auto_layout(nodes_data, root_id):
    positions = {}
    
    function get_children(node_id):
        return nodes_data[node_id]["children"]
    
    function calc_subtree_width(node_id):
        children = get_children(node_id)
        if children is empty:
            return NODE_WIDTH
        total = sum(calc_subtree_width(c) for c in children)
        return max(total, NODE_WIDTH)
    
    function layout_node(node_id, x, y):
        positions[node_id] = (x, y)
        children = get_children(node_id)
        if children is not empty:
            total_width = sum(calc_subtree_width(c) for c in children)
            current_x = x - total_width / 2
            for child in children:
                child_width = calc_subtree_width(child)
                layout_node(child, current_x + child_width/2, y + NODE_HEIGHT + V_GAP)
                current_x += child_width
    
    layout_node(root_id, 400, 100)
    return positions
```

### 3.12 项目导入算法

```
function import_project(zip_path, target_dir):
    // Step 1: 验证 ZIP 文件
    if not os.path.exists(zip_path):
        return (false, "文件不存在", null)
    
    if not zipfile.is_zipfile(zip_path):
        return (false, "不是有效的 ZIP 文件", null)
    
    // Step 2: 验证项目结构
    with zipfile.ZipFile(zip_path, 'r') as zipf:
        namelist = zipf.namelist()
        
        if namelist is empty:
            return (false, "ZIP 文件为空", null)
        
        found_project_json = false
        found_tree_json = false
        
        for name in namelist:
            basename = os.path.basename(name)
            if basename == "project.json":
                found_project_json = true
            elif basename == "tree.json":
                found_tree_json = true
        
        if not found_project_json:
            return (false, "缺少 project.json 文件", null)
        
        if not found_tree_json:
            return (false, "缺少 tree.json 文件", null)
    
    // Step 3: 获取项目名称
    project_name = get_project_name(zip_path)
    if not project_name:
        project_name = os.path.splitext(os.path.basename(zip_path))[0]
    
    project_root = os.path.join(target_dir, project_name)
    
    // Step 4: 检查同名冲突
    if os.path.exists(project_root):
        return (false, "PROJECT_EXISTS", project_root)
    
    // Step 5: 解压项目
    try:
        os.makedirs(target_dir, exist_ok=True)
        
        with zipfile.ZipFile(zip_path, 'r') as zipf:
            zip_root = get_project_root_in_zip(zip_path)
            
            for member in zipf.namelist():
                if zip_root:
                    if not member.startswith(zip_root + "/"):
                        continue
                    relative_path = member[len(zip_root) + 1:]
                else:
                    relative_path = member
                
                if not relative_path or relative_path.endswith("/"):
                    continue
                
                target_path = os.path.join(project_root, relative_path)
                os.makedirs(os.path.dirname(target_path), exist_ok=True)
                
                with zipf.open(member) as source:
                    with open(target_path, 'wb') as target:
                        target.write(source.read())
        
        return (true, "", project_root)
        
    except Exception as e:
        return (false, f"解压失败: {str(e)}", null)

function get_project_root_in_zip(zip_path):
    with zipfile.ZipFile(zip_path, 'r') as zipf:
        for name in zipf.namelist():
            if name.endswith("project.json"):
                parts = name.rsplit("/", 1)
                if len(parts) > 1:
                    return parts[0]
                return ""
    return null

// 生成不冲突的新名称
function generate_new_name(base_name, target_dir):
    if not os.path.exists(os.path.join(target_dir, base_name)):
        return base_name
    
    counter = 1
    while true:
        new_name = f"{base_name}_{counter}"
        if not os.path.exists(os.path.join(target_dir, new_name)):
            return new_name
        counter += 1
        
        if counter > 1000:
            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            return f"{base_name}_{timestamp}"
```

```
function get_dd_code(key_name):
    key_lower = key_name.lower()
    
    // 检查缓存
    if key_lower in vk_cache:
        return vk_cache[key_lower]
    
    // 查找VK码
    vk_code = VK_CODE_MAP.get(key_lower)
    if vk_code is null:
        if len(key_lower) == 1 and key_lower.isalpha():
            vk_code = ord(key_lower.upper())
        else:
            return 0  // 未知按键
    
    // 通过DD DLL转换
    if dd_dll is available:
        dd_code = dd_dll.DD_todc(vk_code)
        if dd_code > 0:
            vk_cache[key_lower] = dd_code  // 缓存结果
            return dd_code
    
    return 0  // 转换失败
```

### 3.12 Threshold 参数兼容性转换算法

```
Algorithm: ThresholdConversion
Input: threshold_value (float or int or str)
Output: normalized_threshold (float, range 0.0-1.0)

function normalize_threshold(threshold_value):
    // 转换为浮点数
    if isinstance(threshold_value, str):
        threshold = float(threshold_value)
    else:
        threshold = float(threshold_value)
    
    // 兼容性转换：百分比格式 -> 小数格式
    // 旧项目 (autodoor) 使用百分比格式，如 65 表示 65%
    // 新项目使用小数格式，如 0.65 表示 65%
    if threshold > 1.0:
        threshold = threshold / 100.0
    
    return threshold

// 示例:
// normalize_threshold(65)    → 0.65  (旧项目百分比格式)
// normalize_threshold("65")  → 0.65  (字符串百分比格式)
```

### 6.2 快捷键绑定表

**重要：快捷键必须绑定到顶级窗口**

使用 `root.bind()` 而非 `self.bind()`，确保在画布获得焦点时快捷键仍能生效：

```
Algorithm: BindShortcuts
Input: shortcuts (List of (key, callback) pairs)

function bind_shortcuts(shortcuts):
    root = get_toplevel()
    
    for key, callback in shortcuts:
        handler = make_handler(callback, key)
        root.bind(key, handler)

function make_handler(callback, key):
    def handler(event):
        if key in ["<Delete>", "<BackSpace>"]:
            focused = get_toplevel().focus_get()
            if focused:
                widget_type = get_class_name(focused)
                if widget_type in ["CTkEntry", "Entry", "CTkTextbox", "Text"]:
                    return None  // 允许输入控件正常处理
        
        if is_callable(callback):
            callback()
        return "break"  // 阻止事件继续传播
    
    return handler
```

### 8.4 图像特征分析算法 (v1.3.0 新增)

```
Algorithm: AnalyzeImageFeatures
Input: image (PIL.Image)
Output: ImageFeatures

function analyze_image_features(image):
    // 转换为灰度图
    gray = cvt_color(image, RGB2GRAY)
    
    // 尺寸特征
    height, width = gray.shape
    min_dimension = min(width, height)
    
    // 亮度特征
    mean_brightness = mean(gray)
    brightness_std = std(gray)
    
    // 对比度特征（Michelson对比度）
    min_val = min(gray)
    max_val = max(gray)
    contrast = (max_val - min_val) / (max_val + min_val + epsilon)
    is_low_contrast = contrast < LOW_CONTRAST_THRESHOLD  // 0.3
    
    // 噪点特征（拉普拉斯方差）
    laplacian = laplacian(gray)
    noise_level = variance(laplacian) / 10000
    has_noise = noise_level > NOISE_THRESHOLD  // 0.5
    
    // 字符高度估计（轮廓检测）
    binary = threshold(gray, 0, 255, THRESH_BINARY + THRESH_OTSU)
    contours = find_contours(binary, RETR_EXTERNAL)
    heights = [bounding_rect(c).height for c in contours if height > 5]
    estimated_char_height = median(heights) if heights else 20
    is_small_font = estimated_char_height < SMALL_FONT_HEIGHT_THRESHOLD  // 15
    
    // 背景特征
    is_dark_background = mean_brightness < DARK_BACKGROUND_THRESHOLD  // 128
    
    // 渐变背景检测
    edges = canny(gray, 50, 150)
    edge_density = sum(edges > 0) / (width * height)
    has_gradient = edge_density < 0.05 and brightness_std > 30
    
    return ImageFeatures(
        width, height, min_dimension,
        mean_brightness, brightness_std,
        contrast, is_low_contrast,
        noise_level, has_noise,
        estimated_char_height, is_small_font,
        is_dark_background, has_gradient
```

### 8.7 精确关键词定位算法 (v1.3.0 新增)

```
Algorithm: CalculateKeywordPosition
Input: text (string), keyword (string), box (bounding box)
Output: position (x, y)

function calculate_keyword_position(text, keyword, box):
    // 字符宽度系数表
    CHAR_WIDTH_FACTORS = {
        'chinese': 2.0,
        'fullwidth': 2.0,
        'uppercase': 1.2,
        'lowercase': 1.0,
        'digit': 1.0,
        'space': 0.5,
        'punctuation': 0.6
    }
    
    // 查找关键词位置
    keyword_idx = find(text.lower(), keyword.lower())
    if keyword_idx == -1:
        // 返回文本框中心
        return (box.center_x, box.center_y)
    
    // 计算每个字符的相对宽度
    char_widths = [get_char_width_factor(char) for char in text]
    total_width = sum(char_widths)
    
    // 计算关键词起始位置的累积宽度
    start_width = sum(char_widths[0:keyword_idx])
    
    // 计算关键词本身的宽度
    keyword_widths = [get_char_width_factor(char) for char in keyword]
    keyword_width = sum(keyword_widths)
    
    // 计算关键词中心的相对位置
    center_width = start_width + keyword_width / 2
    center_ratio = center_width / total_width
    
    // 转换为像素坐标
    x = int(box.left + box.width * center_ratio)
    y = int(box.top + box.height / 2)
    
    return (x, y)


function get_char_width_factor(char):
    if is_chinese(char):       // U+4E00 to U+9FFF
        return CHAR_WIDTH_FACTORS['chinese']
    if is_fullwidth(char):     // U+FF00 to U+FFEF
        return CHAR_WIDTH_FACTORS['fullwidth']
    if is_uppercase(char):
        return CHAR_WIDTH_FACTORS['uppercase']
    if is_lowercase(char):
        return CHAR_WIDTH_FACTORS['lowercase']
    if is_digit(char):
        return CHAR_WIDTH_FACTORS['digit']
    if is_space(char):
        return CHAR_WIDTH_FACTORS['space']
    if is_punctuation(char):
        return CHAR_WIDTH_FACTORS['punctuation']
    return 1.0
```

### 11.2 拖拽阈值检测算法

```
Algorithm: OnNodeDrag
Input: event (MouseEvent)
Output: None

function on_drag(event):
    x, y = convert_to_world_coordinates(event)
    
    // 检查是否应该触发拖拽
    if not _dragging and _click_pos and _click_node_id:
        // 检查鼠标按钮是否仍然按下
        mouse_pressed = event.state & 0x0100
        if not mouse_pressed:
            // 鼠标已释放，清除状态
            _click_pos = null
            _click_node_id = null
            return
        
        // 计算移动距离
        dx = x - _click_pos[0]
        dy = y - _click_pos[1]
        distance = sqrt(dx*dx + dy*dy)
        
        // 移动超过阈值才触发拖拽
        if distance >= DRAG_THRESHOLD:
            if _click_node_id in nodes and _click_node_id in selected_nodes:
                _dragging = true
                _drag_node = _click_node_id
                node = nodes[_click_node_id]
                _drag_start = (_click_pos[0] - node.x, _click_pos[1] - node.y)
                _drag_start_pos = (node.x, node.y)
                _drag_start_positions = {}
                for each nid in selected_nodes:
                    if nid in nodes:
                        n = nodes[nid]
                        _drag_start_positions[nid] = (n.x, n.y)
    
    // 执行拖拽逻辑
    if _dragging and _drag_node:
        // 再次检查鼠标按钮状态
        mouse_pressed = event.state & 0x0100
        if not mouse_pressed:
            // 鼠标已释放，清除拖拽状态
            _dragging = false
            _drag_node = null
            _click_pos = null
            _click_node_id = null
            return
        
        // 计算偏移量
        dx = x - _drag_start[0] - _drag_start_pos[0]
        dy = y - _drag_start[1] - _drag_start_pos[1]
        
        // 移动所有选中的节点
        if len(selected_nodes) > 1 and _drag_start_positions:
            for each node_id in selected_nodes:
                if node_id in nodes and node_id in _drag_start_positions:
                    start_x, start_y = _drag_start_positions[node_id]
                    nodes[node_id].move_to(start_x + dx, start_y + dy)
        else:
            node = nodes[_drag_node]
            node.move_to(x - _drag_start[0], y - _drag_start[1])
        
        _redraw_connections()
```

```
function import_resource(source_path, project_root):
    // 1. 检测资源类型
    resource_type = detect_resource_type(source_path)
    
    // 2. 确定目标目录
    target_dir = get_target_directory(resource_type)
    // resource_type → target_dir 映射:
    // "image" → "images/templates"
    // "script" → "scripts/script"
    // "code" → "scripts/code"
    // "audio" → "audio/alarms"
    // "data" → "data/config"
    
    // 3. 处理文件名冲突
    filename = basename(source_path)
    target_path = join(project_root, target_dir, filename)
    
    if exists(target_path):
        counter = 2
        while exists(target_path):
            name, ext = splitext(filename)
            new_filename = f"{name}_{counter}{ext}"
            target_path = join(project_root, target_dir, new_filename)
            counter += 1
        filename = new_filename
    
    // 4. 复制文件到项目目录
    create_directory(dirname(target_path))
    copy_file(source_path, target_path)
    
    // 5. 生成相对路径引用
    relative_path = "./" + target_dir + "/" + filename
    relative_path = relative_path.replace("\\", "/")
    
    return relative_path
```

```
function resolve_path(relative_path, project_root):
    // 1. 检查是否为相对路径
    if not relative_path starts with "./":
        return relative_path  // 已经是绝对路径
    
    // 2. 移除 "./" 前缀
    path_without_prefix = relative_path[2:]
    
    // 3. 构建绝对路径
    absolute_path = join(project_root, path_without_prefix)
    
    // 4. 规范化路径
    absolute_path = normalize_path(absolute_path)
    
    // 5. 验证路径安全性
    real_path = realpath(absolute_path)
    real_project_root = realpath(project_root)
    
    if not real_path starts with real_project_root:
        raise Error("路径超出项目范围")
    
    return absolute_path
```

```
function import_single_file_to_project(source_path, resource_type, old_path=null):
    target_dir = TYPE_DIR_MAP.get(resource_type, 'data/other')
    filename = basename(source_path)
    target_path = join(project_root, target_dir, filename)
    
    // 处理文件名冲突

function move_to_cache(file_path):

        return false

        return false

    create_directory(cache_dir)

    timestamp = format_datetime(now(), "%Y%m%d_%H%M%S")
    cache_filename = f"{name}_{timestamp}{ext}"
    cache_path = join(cache_dir, cache_filename)
    
    move_file(file_path, cache_path)
    return true

function update_tree_paths(tree_data, path_mapping):
    nodes = tree_data.get('nodes', {})
    
    for key, new_path in path_mapping:
        node_id, path_key = rsplit(key, '.', 1)
        if node_id in nodes:
            if 'config' not in nodes[node_id]:
                nodes[node_id]['config'] = {}
            nodes[node_id]['config'][path_key] = new_path
    
    return tree_data
```

**TextListField 说明：**
- 用于 TextInputNode 的 `preset_texts` 属性
- 使用 CTkTextbox 多行文本输入组件
- 每行一条文本，`get_value()` 将文本按行分割返回列表
- `set_value()` 将列表按行拼接后设置到文本框
- 支持焦点边框颜色变化
