```python
# tests/test_screen_utils.py

class TestGetVirtualScreenBounds:
    """测试虚拟屏幕边界计算"""

    def test_returns_four_values(self):
        """返回值应为4元组 (min_x, min_y, max_x, max_y)"""
        from bt_utils.screen_utils import get_virtual_screen_bounds
        bounds = get_virtual_screen_bounds()
        assert isinstance(bounds, tuple)
        assert len(bounds) == 4
        assert all(isinstance(v, int) for v in bounds)

    def test_bounds_are_valid(self):
        """边界值应满足 min < max"""
        from bt_utils.screen_utils import get_virtual_screen_bounds
        min_x, min_y, max_x, max_y = get_virtual_screen_bounds()
        assert min_x <= max_x
        assert min_y <= max_y
        assert (max_x - min_x) > 0
        assert (max_y - min_y) > 0

    def test_single_monitor_default(self):
        """单显示器时边界应合理（正数，不超过常见分辨率）"""
        from bt_utils.screen_utils import get_virtual_screen_bounds
        min_x, min_y, max_x, max_y = get_virtual_screen_bounds()
        assert max_x <= 7680  # 不超过4个4K显示器
        assert max_y <= 4320
        assert (max_x - min_x) >= 800  # 至少800px宽
        assert (max_y - min_y) >= 600  # 至少600px高

    def test_cache_works(self):
        """多次调用应返回相同结果（缓存验证）"""
        from bt_utils.screen_utils import get_virtual_screen_bounds
        b1 = get_virtual_screen_bounds()
        b2 = get_virtual_screen_bounds()
        assert b1 == b2

    def test_fallback_when_screeninfo_unavailable(self):
        """screeninfo 不可用时应 fallback 到 Win32 API 或默认值"""
        import sys
        # 模拟 screeninfo 不可用
        original_modules = sys.modules.copy()
        try:
            sys.modules['screeninfo'] = None
            # 清除缓存
            import bt_utils.screen_utils as su
            su._virtual_screen_bounds_cache = None
            bounds = su.get_virtual_screen_bounds()
            assert isinstance(bounds, tuple)
            assert len(bounds) == 4
        finally:
            sys.modules.clear()
            sys.modules.update(original_modules)


class TestGetVirtualScreenOffset:
    """测试虚拟屏幕偏移量"""

    def test_returns_two_values(self):
        """返回值应为2元组 (offset_x, offset_y)"""
        from bt_utils.screen_utils import get_virtual_screen_offset
        offset = get_virtual_screen_offset()
        assert isinstance(offset, tuple)
        assert len(offset) == 2

    def test_single_monitor_offset_is_zero(self):
        """单显示器时偏移量应为 (0, 0)"""
        from bt_utils.screen_utils import get_virtual_screen_offset
        offset_x, offset_y = get_virtual_screen_offset()
        # 单屏时 min_x=0, min_y=0, 所以 offset 应为 (0, 0)
        # 多屏时可能为负值
        assert isinstance(offset_x, int)
        assert isinstance(offset_y, int)

    def test_offset_consistent_with_bounds(self):
        """偏移量应与边界计算的 min 值一致"""
        from bt_utils.screen_utils import get_virtual_screen_bounds, get_virtual_screen_offset
        min_x, min_y, _, _ = get_virtual_screen_bounds()
        offset_x, offset_y = get_virtual_screen_offset()
        assert offset_x == -min_x
        assert offset_y == -min_y
```

```python
# 验证不再有直接 ImageGrab.grab 调用（除 screen_service 内部）
def test_no_bypass_calls():
    """检查 widgets.py 和 magnifier.py 不再直接调用 ImageGrab.grab"""
    import ast
    import inspect

    # 检查 widgets.py
    import bt_gui.widgets
    source = inspect.getsource(bt_gui.widgets)
    tree = ast.parse(source)
    for node in ast.walk(tree):
        if isinstance(node, ast.Call):
            if isinstance(node.func, ast.Attribute):
                if (isinstance(node.func.value, ast.Name) and
                    node.func.value.id == 'ImageGrab' and
                    node.func.attr == 'grab'):
                    assert False, "widgets.py 仍存在直接 ImageGrab.grab 调用"

    # 检查 magnifier.py
    import bt_utils.magnifier
    source = inspect.getsource(bt_utils.magnifier)
    tree = ast.parse(source)
    for node in ast.walk(tree):
        if isinstance(node, ast.Call):
            if isinstance(node.func, ast.Attribute):
                if (isinstance(node.func.value, ast.Name) and
                    node.func.value.id == 'ImageGrab' and
                    node.func.attr == 'grab'):
                    assert False, "magnifier.py 仍存在直接 ImageGrab.grab 调用"
```

```python
import sys

def verify_p0():
    """P0 阶段验证"""
    print("=" * 50)
    print("P0: 验证 screen_utils.py")
    print("=" * 50)
    
    try:
        from bt_utils.screen_utils import get_virtual_screen_bounds, get_virtual_screen_offset
        bounds = get_virtual_screen_bounds()
        offset = get_virtual_screen_offset()
        
        print(f"  虚拟屏幕边界: {bounds}")
        print(f"  虚拟屏幕偏移: {offset}")
        
        assert len(bounds) == 4, "边界应为4元组"
        assert len(offset) == 2, "偏移应为2元组"
        assert bounds[0] <= bounds[2], "min_x <= max_x"
        assert bounds[1] <= bounds[3], "min_y <= max_y"
        assert offset[0] == -bounds[0], "offset_x == -min_x"
        assert offset[1] == -bounds[1], "offset_y == -min_y"
        
        print("  ✅ P0 验证通过")
        return True
    except Exception as e:
        print(f"  ❌ P0 验证失败: {e}")
```
